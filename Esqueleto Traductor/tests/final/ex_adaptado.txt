    if (a ^ b) {
        print 'T';
        print '\n';
    }
    else {
        print 'F';
        print '\n';      //   # False
    }
    if (a && check(b)){           //                # Check
        print 'T';
        print '\n';         //    # True
    } else {
        print 'F';
        print '\n';
    }

    if (!a && check(b)) {
        print 'T';
        print '\n';
    } else {
        print 'F';
        print '\n';   //  # False
    }
}
